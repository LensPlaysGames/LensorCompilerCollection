* Bug 001

As you can see, the generated assembly contains (hopefully used to
contain) no return instruction, meaning the computer starts executing
random data in the memory following the last =add= instruction.

#+begin_src asm
main:
    push %rbp
    mov %rsp, %rbp
    sub $32, %rsp
.Lbody_0:
    mov %edi, -4(%rbp)
    mov %rsi, -12(%rbp)
    mov %rdx, -20(%rbp)
    mov $0, %rax
    mov %rbp, %rsp
    pop %rbp
    ret

_XGlintfooF0_B5:
    push %rbp
    mov %rsp, %rbp
    mov $1, %rax
    add $1, %rax
#+end_src

semantic analysis from properly inserting a return instruction.
